    try:
        file = open(file_path,"r")

            raise CustomeExceptionForEmptyFile("THE FILE IS EMPTY YOU MUSTT READ UNEMPTY FILE")

            raise CustomeExceptionForBigFile("THIS FILE IS BIGGER THAN 10 mb YOU MUSTT READ SMALLER FILE")

        print("WOOHOOOOO !!")

    except CustomeExceptionForBigFile as ex:
        print(ex) 
    except Exception as ex:
